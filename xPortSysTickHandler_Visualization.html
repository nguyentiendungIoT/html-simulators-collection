FreeRTOS: xPortSysTickHandler()

First Principles: Từ tiếng "KENG" đến Context Switch

portable/GCC/ARM_CM4F/port.c code 
void xPortSysTickHandler( void )
{
    portDISABLE_INTERRUPTS();  // (1) Khóa cửa
    traceISR_ENTER();
    {
        if( xTaskIncrementTick() != pdFALSE ) // (2) Tim đập & Kiểm tra
        {
            traceISR_EXIT_TO_SCHEDULER();
            portNVIC_INT_CTRL_REG = portNVIC_PENDSVSET_BIT; // (3) Cắm cờ
        }
        else
        {
            traceISR_EXIT();
        }
    }
    portENABLE_INTERRUPTS(); // Mở cửa
}
